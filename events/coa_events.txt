# Coat of Arms events

namespace = coa

coa.1 = {
	type = character_event
	hidden = yes
	trigger = {
		is_ruler = yes
	}
	immediate = {
		trigger_event = coa.2 # Achaean League
	}
}

coa.2 = {
	type = character_event
	hidden = yes
	trigger = {
		any_held_title = { this = title:d_achaia }
	}
	immediate = {
		if = {
			limit = { 
				capital_county = { this = title:c_kerynaia }
			}
			title:d_achaia = { set_coa = d_achaia_kerynaia }
		}
		else_if = {
			limit = { 
				capital_county = { this = title:c_dyme }
			}
			title:d_achaia = { set_coa = d_achaia_dyme }
		}
		else_if = {
			limit = { 
				capital_county = { this = title:c_patrai }
			}
			title:d_achaia = { set_coa = d_achaia_patrai }
		}
		else = {
			title:d_achaia = { set_coa = d_achaia }
		}
	}
}
